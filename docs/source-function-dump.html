 62:  * Insert XML into a SimpleXMLElement
 63:  * @from http://stackoverflow.com/questions/767327/in-simplexml-how-can-i-add-an-existing-simplexmlelement-as-a-child-element
 64:  * @param SimpleXMLElement $parent
 65:  * @param string $xml
 66:  * @param bool $before
 67:  * @return bool XML string added
 68:  */
 69: function simplexml_import_xml(SimpleXMLElement $parent, $xml, $before = false)
 70: {
 71:     $xml = (string)$xml;
 72: 
 73:     // check if there is something to add
 74:     if ($nodata = !strlen($xml) or $parent[0] == NULL) {
 75:         return $nodata;
 76:     }
 77: 
 78:     // add the XML
 79:     $node = dom_import_simplexml($parent);
 80:     $fragment = $node->ownerDocument->createDocumentFragment();
 81:     $fragment->appendXML($xml);
 82: 
 83:     if ($before) {
 84:         return (bool)$node->parentNode->insertBefore($fragment, $node);
 85:     }
 86: 
 87:     return (bool)$node->appendChild($fragment);
 88: }
 89: /**
 90:  * Insert SimpleXMLElement into SimpleXMLElement
 91:  * @from http://stackoverflow.com/questions/767327/in-simplexml-how-can-i-add-an-existing-simplexmlelement-as-a-child-element
 92:  * @param SimpleXMLElement $parent
 93:  * @param SimpleXMLElement $child
 94:  * @param bool $before
 95:  * @return bool SimpleXMLElement added
 96:  */
 97: function simplexml_import_simplexml(SimpleXMLElement $parent, SimpleXMLElement $child, $before = false)
 98: {
 99:     // check if there is something to add
100:     if ($child[0] == NULL) {
101:         return true;
102:     }
103: 
104:     // if it is a list of SimpleXMLElements default to the first one
105:     $child = $child[0];
106: 
107:     // insert attribute
108:     if ($child->xpath('.') != array($child)) {
109:         $parent[$child->getName()] = (string)$child;
110:         return true;
111:     }
112: 
113:     $xml = $child->asXML();
114: 
115:     // remove the XML declaration on document elements
116:     if ($child->xpath('/*') == array($child)) {
117:         $pos = strpos($xml, "\n");
118:         $xml = substr($xml, $pos + 1);
119:     }
120: 
121:     return simplexml_import_xml($parent, $xml, $before);
122: }
